# 三.相机移动物体

:::demo

```vue
<template>
  <div ref="example"></div>
</template>

<script>
export default {
  methods: {
    init() {
      // 对象
      var geometry = new THREE.CylinderGeometry(100, 150, 400)
      // 材质
      var material = new THREE.MeshLambertMaterial({ color: 0xffaa00 })
      // 模型
      var mesh = new THREE.Mesh(geometry, material)
      mesh.position = new THREE.Vector3(0, 0, 0)
      // 场景
      var scene = new THREE.Scene()
      var objs = new THREE.Object3D()
      var axesHelper = new THREE.AxesHelper(800)
      objs.position.z = -1000
      objs.add(mesh)
      objs.add(axesHelper)
      new TWEEN.Tween(objs.rotation)
        .to({ y: 360, z: 360, x: 360 }, 1000000)
        .repeat(Infinity)
        .start()
      scene.add(objs)
      var paramObj = function () {
        this.fov = 45
      }
      var param = new paramObj()
      var gui = new dat.GUI()
      gui.add(param, "fov", 0, 180).name("视角大小")
      // 灯光
      var light = new THREE.AmbientLight(0xff0000)
      light.position.set(100, 100, 200)
      scene.add(light)
      light = new THREE.PointLight(0x00ff00)
      light.position.set(0, 0, 300)
      scene.add(light)
      // 相机
      var camera = new THREE.PerspectiveCamera(
        45,
        window.innerWidth / window.innerHeight,
        1,
        10000
      )
      camera.position.x = 0
      camera.position.y = 0
      camera.position.z = 600
      camera.up.x = 0
      camera.up.y = 1
      camera.up.z = 0
      camera.lookAt(0, 0, 0)
      // 渲染器
      var render = new THREE.WebGLRenderer({
        antialias: true,
      })
      render.setSize(window.innerWidth / 2, window.innerHeight / 2)
      render.setClearColor(0xffffff, 1.0)
      render.render(scene, camera)
      // state
      var stats = new Stats()
      stats.showPanel(0)
      stats.domElement.style.position = "absolute"
      stats.domElement.style.left = "0px"
      stats.domElement.style.bottom = "0px"
      // 挂载
      this.$refs.example.appendChild(render.domElement)
      this.$refs.example.appendChild(stats.domElement)
      // 动画
      function animate() {
        stats.begin()
        camera.fov = param.fov
        camera.updateProjectionMatrix()
        TWEEN.update()
        render.render(scene, camera)
        stats.end()
        requestAnimationFrame(animation)
      }
      animate()
      //窗口尺寸自适应
      window.onresize = function () {
        render.setSize(window.innerWidth / 2, window.innerHeight / 2)
        camera.aspect = window.innerWidth / window.innerHeight
        camera.updateProjectionMatrix()
      }
    },
  },
  mounted() {
    setTimeout(() => {
      this.init()
    }, 5000)
  },
}
</script>
```
